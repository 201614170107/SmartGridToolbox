213  /// @brief Soonest/smallest rank goes first. If equal rank and time, then sort on the id.
214  class ScheduledUpdatesCompare
215  {
216  public:
217  bool operator()(const std::pair<SimCompPtr, Time>& lhs,
218  const std::pair<SimCompPtr, Time>& rhs)
219  {
220  return ((lhs.second < rhs.second) ||
221  (lhs.second == rhs.second && lhs.first->rank() < rhs.first->rank()) ||
222  (lhs.second == rhs.second && lhs.first->rank() == rhs.first->rank() &&
223  (lhs.first->id() < rhs.first->id())));
224  }
225  };

227  // Smallest rank goes first. If equal rank, then sort on the id.
228  class ContingentUpdatesCompare
229  {
230  public:
231  bool operator()(const SimCompPtr lhs, const SimCompPtr rhs)
232  {
233  return ((lhs->rank() < rhs->rank()) ||
234  ((lhs->rank() == rhs->rank()) && (lhs->id() < rhs->id())));
235  }
236  };
